def print_top_words(model, feature_names, n_top_words):
    for topic_idx, topic in enumerate(model.components_):
        print("Topic #%d:" % topic_idx)
        print(" ".join([feature_names[i]                        
                     for i in topic.argsort()[:-n_top_words - 1:-1]]))
    print()

def nearest_neightbor(isrc, tf, labels, top_n=0, scheme="L1"):  
    dists = dict()
    x = labels.loc[labels['isrcs'] == isrc]["isrcs"].index[0]
    for i in range(tf.shape[0]):
        if scheme == "L1":
            dist = np.sum(np.abs(tf[x]- tf[i]), axis = 0)
        elif scheme == "L2":
            dist = np.sqrt(np.sum(np.square(tf[x]- tf[i]), axis = 0))
        dists[labels['isrcs'][i:i+1][i]] = dist
    sorted_dists = OrderedDict(sorted(dists.items(), key=operator.itemgetter(1)))
    return sorted_dists
